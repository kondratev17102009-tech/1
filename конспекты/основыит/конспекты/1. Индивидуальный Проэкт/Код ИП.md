    def draw_wheel(self):

        self.wheel_canvas.delete("all")

        center_x, center_y = 200, 200

        radius = 150

  

        colors = ['#e74c3c', '#3498db', '#2ecc71', '#f1c40f',

                  '#9b59b6', '#e67e22', '#1abc9c', '#e84342',

                  '#6c5ce7', '#00b894']

  

        for i in range(10):

            start_angle = i * 36

            ext_angle = 36

  

            self.wheel_canvas.create_arc(

                center_x - radius, center_y - radius,

                center_x + radius, center_y + radius,

                start=start_angle, extent=ext_angle,

                fill=colors[i], outline='white', width=2

            )

  

            angle = math.radians(start_angle + 18)

            text_x = center_x + radius * 0.7 * math.cos(angle)

            text_y = center_y - radius * 0.7 * math.sin(angle)

            self.wheel_canvas.create_text(

                text_x, text_y,

                text=str(i+1),

                font=('Arial', 16, 'bold'),

                fill='white'

            )

  

        self.wheel_canvas.create_polygon(

            [190, 50, 210, 50, 200, 20],

            fill='white', outline='black', width=2

        )

  

    def spin_wheel(self):

        result_index = random.randint(0, 9)

        self.animate_wheel_spin(result_index)

  

    def animate_wheel_spin(self, result_index):

        spins = random.randint(20, 30)

        current_spin = 0

  

        def spin_step():

            nonlocal current_spin

            if current_spin < spins:

                self.wheel_canvas.delete("all")

                center_x, center_y = 200, 200

                radius = 150

  

                colors = ['#e74c3c', '#3498db', '#2ecc71', '#f1c40f',

                          '#9b59b6', '#e67e22', '#1abc9c', '#e84342',

                          '#6c5ce7', '#00b894']

  

                offset = current_spin * 10

                for i in range(10):

                    start_angle = (i * 36 + offset) % 360

                    ext_angle = 36

  

                    self.wheel_canvas.create_arc(

                        center_x - radius, center_y - radius,

                        center_x + radius, center_y + radius,

                        start=start_angle, extent=ext_angle,

                        fill=colors[i], outline='white', width=2

                    )

  

                self.wheel_canvas.create_polygon(

                    [190, 50, 210, 50, 200, 20],

                    fill='white', outline='black', width=2

                )

  

                current_spin += 1

                self.wheel_window.after(50, spin_step)

            else:

                self.show_wheel_result(result_index)

  

        spin_step()

    def click_barista(self, event):

        if self.energy > 0 or self.debug_mode:

            gain = self.click_power

  

            if random.random() * 100 < (self.double_upgrade_level * 10):

                gain *= 2

                self.create_particle(event.x, event.y, "ДВОЙНОЙ!", '#f39c12')

  

            if random.random() * 100 < self.crit_chance:

                gain *= 2

                self.create_particle(event.x, event.y, "КРИТ!", '#e74c3c')

  

            self.coins += gain

            self.xp += 1

            if not self.debug_mode:

                self.energy -= 5

  

            self.update_ui()

            self.create_coin_effect(event.x, event.y)

            self.shake_barista()

  

    def create_particle(self, x, y, text, color):

        particle = self.canvas.create_text(x, y-30, text=text,

                                           font=('Arial', 12, 'bold'), fill=color)

        self.particles.append({

            'id': particle,

            'x': x,

            'y': y-30,

            'vy': -2,

            'life': 30

        })

    def shake_barista(self):

        if hasattr(self, 'barista'):

            for _ in range(3):

                dx = random.randint(-5, 5)

                dy = random.randint(-3, 3)

                self.canvas.move(self.barista, dx, dy)

                self.window.after(

                    50,

                    lambda dx=dx, dy=dy: self.canvas.move(self.barista, -dx, -dy)

                )

            text="💰",

            font=('Arial', 20),

            fill='gold'

            'y': y,

            'vy': -3,

            'life': 20

        })

  

    def animate_particles(self):

        for particle in self.particles[:]:

            particle['life'] -= 1

            if particle['life'] <= 0:

                self.canvas.delete(particle['id'])

                self.particles.remove(particle)

            else:

                particle['y'] += particle['vy']

                self.canvas.coords(particle['id'], particle['x'], particle['y'])

  

        self.window.after(50, self.animate_particles)

  

    def update_energy(self):

        if self.energy < self.max_energy and not self.debug_mode:

            self.energy += self.energy_regen

            if self.energy > self.max_energy:

                self.energy = self.max_energy

  

        width = 300 * (self.energy / self.max_energy)

        self.canvas.coords(self.energy_bar, 450, 25, 450 + width, 55)

        self.canvas.itemconfig(self.energy_text,

                               text=f"⚡ {int(self.energy)}/{self.max_energy}")

  

        self.window.after(1000, self.update_energy)
